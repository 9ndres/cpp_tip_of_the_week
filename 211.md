</p></details><details open><summary>Example</summary><p>

```cpp
int main() {
  auto value = 42;

  inspect (value) {
    0  => { std::cout << 0; }
    42 => { std::cout << 42; } // prints 42
    x  => { std::cout << x; }
  };

  auto tuple = std::tuple{"Quantlab", 42};

  inspect(tuple) {
    [str, 42] => { std::cout << str << ' ' << 42; } // prints Quantlab42
    [_, _]    => { std::cout << "..."; }
  };

  return inspect(value) {
    _ => _ + 1; // returns 43
  };
}
```
